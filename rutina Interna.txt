DEFINE DATA
  LOCAL
    1 V-CONTRATO VIEW OF NCCRUISE
      2 CRUISE-ID (N8)
      2 DESTINATION-HARBOR (A20)
END-DEFINE

READ (5) V-CONTRATO LOGICAL BY V-CONTRATO.CRUISE-ID

  PERFORM MOSTRAR-PUERTO

  SKIP 1

END-READ

END

DEFINE SUBROUTINE MOSTRAR-PUERTO
  WRITE 'PUERTO DESTINO: ' V-CONTRATO.DESTINATION-HARBOR
END-SUBROUTINE
